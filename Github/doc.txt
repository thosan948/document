// Clone source từ Github
    git clone "link_source"
    
// Tạo Branch mới
    git checkout -b "name_branch"

// Clear file
    git checkout

// Xác định vị trí Branch hiện tại
    git branch

// Chuyển sang 1 Branch khác
    git checkout "name_branch"

// Xem thay đổi của Branch
    git status

// Add những thứ thay đổi.
    git add .

// Commit
    git commit -m "status"

// Push code của Branch lên Github
    git push origin "name_branch"

// Merge code từ Branch khác vào Branch hiện tại
    git merge "name_branch"
